29 valeursPID() : Kp(0.0), Ki(0.0), Kd(0.0), initialTime(0), Sp(0.0), Pv(0.0), integral(0.0), previous_error(0.0), Out(0.0) {}
30 float Kp; // Constante proportionnelle
31 float Ki; // Constante intégrale
32 float Kd; // Constante dérivée
33 long initialTime; // Temps initial
34 float Sp; // Set Point (Valeur voulue)
35 float Pv; // Process Value (Valeur réelle)
36 float integral; // Valeur intégrale
37 float previous_error; // Previous error value
38 float Out; // Valeur de sortie
39 float update() {
40 unsigned long currentTime = millis(); // Get current time in milliseconds
41
42 // Calculate the time difference (dt) since the last update
43 float dt = (currentTime - initialTime) / 1000.0; // Convert to seconds
44
45 // Calculate the error
46 float error = Sp - Pv;
47
48 // Update the integral term
49 integral += error * dt;
50
51 // Calculate the derivative term
52 float derivative = (error - previous_error) / dt;
53
54 // Calculate the control output
55 Out = Kp * error + Ki * integral + Kd * derivative;
56
57 // Store the current time for the next update
58 initialTime = currentTime;
59
60 // Store the current error for the next iteration
61 previous_error = error;
62
63 return Out;
64 }
